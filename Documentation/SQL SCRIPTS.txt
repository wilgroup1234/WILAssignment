CREATE TABLE Users
(
    UserID int NOT NULL IDENTITY PRIMARY KEY,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255) NOT NULL,
    Age int NOT NULL,
	Email varchar(255) UNIQUE NOT NULL,
	Password varchar(255) NOT NULL 
 
); 


CREATE TABLE Goals
(
    GoalID int NOT NULL IDENTITY PRIMARY KEY,
    GoalName varchar(255) NOT NULL,
    GoalDescription varchar(255) NOT NULL
); 

CREATE TABLE CustomGoals
(
    GoalID int NOT NULL IDENTITY PRIMARY KEY,
    GoalName varchar(255) NOT NULL,
    GoalDescription varchar(255) NOT NULL
); 


CREATE TABLE DailyQuote
(
    QuoteID int NOT NULL IDENTITY PRIMARY KEY,
    QuoteDate Date NOT NULL,
	QuoteText varchar(255) NOT NULL,
    YoutubeLink varchar(255) NOT NULL,
	TemplateID int NOT NULL
); 


CREATE TABLE Template
(
    TemplateID int NOT NULL IDENTITY PRIMARY KEY,
    TemplateName Varchar(255) NOT NULL
); 


CREATE TABLE UserGoals
(
    UserGoalID int NOT NULL IDENTITY PRIMARY KEY,
    UserID int NOT NULL,
    GoalID int NOT NULL,
    Completed int NOT NULL
); 


CREATE TABLE CustomUserGoals
(
    UserGoalID int NOT NULL IDENTITY PRIMARY KEY,
    UserID int NOT NULL,
    GoalID int NOT NULL,
    Completed int NOT NULL
); 

CREATE TABLE LifeSkills
(
    LifeSkillID int NOT NULL IDENTITY PRIMARY KEY,
    LifeSkillName Varchar(255) NOT NULL
); 

CREATE TABLE CVs
(
    CVID int NOT NULL IDENTITY PRIMARY KEY,
    LifeSkillName Varchar(255) NOT NULL,
	IDNumber Varchar(255) NOT NULL,
    DateOfBIrth Date NOT NULL,
	HighSchoolName Varchar(255) NOT NULL,
	PreviousWorkExperience Varchar(255) NOT NULL,
	WorkReferences Varchar(255) NOT NULL,
	Languages Varchar(255) NOT NULL,
	Achievements Varchar(255) NOT NULL,
	Nationality Varchar(255) NOT NULL,
	Interests Varchar(255) NOT NULL,
	Email Varchar(255) NOT NULL,
	PhoneNumber Varchar(255) NOT NULL,
	Address Varchar(255) NOT NULL
); 

CREATE TABLE UserLifeSkills
(
    UserLifeSkillID int NOT NULL IDENTITY PRIMARY KEY,
    UserID int NOT NULL,
    LifeSKillID int NOT NULL,
    Completed int NOT NULL
); 

CREATE TABLE UserCVs
(
    UserCVID int NOT NULL IDENTITY PRIMARY KEY,
    UserID int NOT NULL,
    CVID int NOT NULL
);

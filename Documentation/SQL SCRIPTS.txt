CREATE TABLE Users
(
    UserID int NOT NULL IDENTITY PRIMARY KEY,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255) NOT NULL,
    Email varchar(255) UNIQUE NOT NULL,
    Password varchar(255) NOT NULL 
 
); 


CREATE TABLE Goals
(
    GoalID int NOT NULL IDENTITY PRIMARY KEY,
    GoalName varchar(255) NOT NULL,
    GoalDescription varchar(255) NOT NULL
); 

CREATE TABLE CustomGoals
(
    GoalID int NOT NULL IDENTITY PRIMARY KEY,
    GoalName varchar(255) NOT NULL,
    GoalDescription varchar(255) NOT NULL,
    FinishDate Date NULL
); 


CREATE TABLE DailyQuote
(
    QuoteID int NOT NULL IDENTITY PRIMARY KEY,
    QuoteDate Date NOT NULL,
	QuoteText varchar(255) NOT NULL,
                YoutubeLink varchar(255) NOT NULL,
	TemplateID int NOT NULL,
               Views int NOT NULL
); 


CREATE TABLE Template
(
    TemplateID int NOT NULL IDENTITY PRIMARY KEY,
    TemplateName Varchar(255) NOT NULL
); 


CREATE TABLE UserGoals
(
    UserGoalID int NOT NULL IDENTITY PRIMARY KEY,
    UserID int NOT NULL,
    GoalID int NOT NULL,
    Completed int NOT NULL
); 


CREATE TABLE CustomUserGoals
(
    UserGoalID int NOT NULL IDENTITY PRIMARY KEY,
    UserID int NOT NULL,
    GoalID int NOT NULL,
    Completed int NOT NULL
); 

CREATE TABLE LifeSkills
(
    LifeSkillID int NOT NULL IDENTITY PRIMARY KEY,
    LifeSkillName Varchar(255) NOT NULL
); 


CREATE TABLE UserLifeSkills
(
    UserLifeSkillID int NOT NULL IDENTITY PRIMARY KEY,
    UserID int NOT NULL,
    LifeSKillID int NOT NULL,
    Completed int NOT NULL
); 


CREATE TABLE PasswordReset
(
    id int NOT NULL IDENTITY PRIMARY KEY,
	passwordCode varchar(255) NOT NULL 
); 


CREATE TABLE Streaks
(
    StreakID int NOT NULL IDENTITY PRIMARY KEY,
    UserID int NOT NULL,
    StreakLength int NOT NULL
); 

CREATE TABLE UserLoginDates
(
    UserLoginDatesID int NOT NULL IDENTITY PRIMARY KEY,
    UserID int NOT NULL,
    UserLoginDate Date NOT NULL
 
); 

CREATE TABLE UserSteps
(
    UserStepsID int NOT NULL IDENTITY PRIMARY KEY,
    UserStepsDate Date NOT NULL,
    Steps int UNIQUE NOT NULL,
    UserID int NOT NULL 
 
); 

CREATE TABLE Gratitude
(
    GratitudeID int NOT NULL IDENTITY PRIMARY KEY,
    GratitudeDate Date NOT NULL,
    GratitudeItems varchar(255) NOT NULL,
    UserID int NOT NULL
 
); 

CREATE TABLE Leaderboards
(
    ID int NOT NULL IDENTITY PRIMARY KEY,
    Score int NOT NULL,
    UserID int NOT NULL
); 

CREATE TABLE SecurityQuestions
(
    ID int NOT NULL IDENTITY PRIMARY KEY,
    Question Varchar(255) NOT NULL,
    Answer Varchar(255) NOT NULL,
    UserID int NOT NULL
);
